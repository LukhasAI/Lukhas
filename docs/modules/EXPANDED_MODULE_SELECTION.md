Create `analyze_all.sh`:

```bash
#!/bin/bash
# Bulk ML Integration Analysis

MODULES=(
    "consciousness/states/async_client.py"
    "consciousness/systems/lambda_mirror.py"
    "core/agi/autonomous_learning.py"
    "core/agi/self_improvement.py"
    "ai_orchestration/lukhas_ai_orchestrator.py"
    "memory/core/unified_memory_orchestrator.py"
    "vivox/moral_alignment/vivox_mae_core.py"
    "governance/ethics/constitutional_ai.py"
    # Add all modules here...
)

mkdir -p reports

for module in "${MODULES[@]}"; do
    echo "Analyzing: $module"
    filename=$(basename "$module" .py)

    python3 ml_integration_analyzer.py \
        --orphan "$module" \
        --lukhas . \
        --cheap-first \
        --extra-info \
        --cascade-models gpt-3.5-turbo,gpt-4,gpt-4-turbo \
        --out "reports/${filename}_analysis.json" \
        2>&1 | tee "reports/${filename}_log.txt"

    echo "Completed: $module"
    echo "---"
    sleep 2  # Rate limiting
done

echo "All analyses complete!"
```
