4) Lightweight Guard (Python) (drop in /enforcement/tone/author_reference_guard.py)

import re
from pathlib import Path
import yaml

BLOCK_CTX_FLAG = "context: academic"

def load_blocklist(p=Path("tone/tools/author_blocklist.yaml")):
    with p.open() as f:
        return yaml.safe_load(f)

def is_academic(file_text:str, file_path:str, cfg)->bool:
    if BLOCK_CTX_FLAG in file_text:
        return True
    return any(str(file_path).startswith(prefix) for prefix in cfg["exceptions"]["paths"])

def scan_text(text, blocked):
    hits=[]
    for term in blocked:
        # case-insensitive whole/near-whole word match
        if re.search(rf"\b{re.escape(term)}\b", text, re.IGNORECASE):
            hits.append(term)
    return sorted(set(hits))

def validate_file(fp:Path, cfg)->list[str]:
    text = fp.read_text(encoding="utf-8", errors="ignore")
    if is_academic(text, str(fp), cfg):
        return []
    hits = scan_text(text, cfg["blocked_terms"])
    return [f"{fp}: blocked reference -> {h}" for h in hits]

def validate_paths(paths):
    cfg = load_blocklist()
    violations=[]
    for root in paths:
        for p in Path(root).rglob("*.*"):
            if p.suffix.lower() in {".md",".mdx",".txt",".py",".tsx",".ts",".js"}:
                violations.extend(validate_file(p, cfg))
    return violations

if __name__ == "__main__":
    import sys
    roots = sys.argv[1:] or ["branding","tone","vocabularies","content"]
    v = validate_paths(roots)
    if v:
        print("‚ùå Author-reference guard violations:")
        for line in v: print(" -", line)
        raise SystemExit(1)
    print("‚úÖ Author-reference guard: clean")

import re, sys

REPLACEMENTS = {
    r"\bKeats(ian)?\b": "poetic yet grounded",
    r"\bMacbeth\b": "tragic grandeur",
    r"\bShakespeare(an)?\b": "classical dramatic",
    r"\bFreud(ian)?\b": "depth-psychology",
    r"\bEinstein(ian)?\b": "cosmic curiosity",
    r"\bZen\b": "attentive presence",
    r"\bTao(ism|ist)?\b": "flow-oriented wisdom",
    r"\bRick Rubin\b": "contemporary creative practice",
    r"\bNachmanovitch\b": "improvisational arts",
    r"\bJulia Cameron\b": "creative coaching"
}

def sanitize(text:str)->str:
    out=text
    for pat, repl in REPLACEMENTS.items():
        out = re.sub(pat, repl, out, flags=re.IGNORECASE)
    return out

if __name__ == "__main__":
    txt=sys.stdin.read()
    print(sanitize(txt))

import re
from pathlib import Path
import yaml

BLOCK_CTX_FLAG = "context: academic"

def load_blocklist(path=Path("tone/tools/author_blocklist.yaml")):
    with path.open(encoding="utf-8") as f:
        return yaml.safe_load(f)

def is_academic(text: str, file_path: str, cfg) -> bool:
    if BLOCK_CTX_FLAG in text:
        return True
    return any(str(file_path).replace("\\", "/").startswith(prefix.strip("/"))
               for prefix in cfg["exceptions"]["paths"])

def scan_text(text: str, blocked: list[str]) -> list[str]:
    hits = []
    # whole-word-ish, case-insensitive
    for term in blocked:
        if re.search(rf"\b{re.escape(term)}\b", text, re.IGNORECASE):
            hits.append(term)
    return sorted(set(hits))

def validate_file(fp: Path, cfg) -> list[str]:
    try:
        text = fp.read_text(encoding="utf-8", errors="ignore")
    except Exception:
        return []
    # gate by extension
    if fp.suffix.lower() not in {".md", ".mdx", ".txt", ".py", ".tsx", ".ts", ".js"}:
        return []
    if is_academic(text, str(fp), cfg):
        return []
    blocked = cfg["blocked_terms"]
    # stance terms are allowed as generic descriptors
    for term in (cfg.get("allow_stance_terms") or []):
        text = text.replace(term, "")  # neutralize before scanning
    hits = scan_text(text, blocked)
    return [f"{fp}: blocked reference -> {h}" for h in hits]

def validate_paths(paths):
    cfg = load_blocklist()
    violations = []
    for root in paths:
        for p in Path(root).rglob("*"):
            violations.extend(validate_file(p, cfg))
    return violations
